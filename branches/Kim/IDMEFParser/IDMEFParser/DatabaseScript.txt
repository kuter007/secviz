create table raw_alert
(
  ID raw(16) default SYS_GUID(),
  MessageID varchar(100) not null,
  AnalyzerID varchar(100) not null,
  CreateTime varchar(100),
  SourceNetworkAddress varchar(100),
  SourcePort number,
  TargetNetworkAddress varchar(100),
  TargetPort number,
  AssessmentImpact varchar(100),
  primary key(ID)
)
